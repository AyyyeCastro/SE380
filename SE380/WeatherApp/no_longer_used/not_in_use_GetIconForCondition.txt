// In the WeatherAPI docs, I could only find reference to the icon .ZIP file.
// So... I downloaded it, and hardcoded icons to the API conditions... I know :( 


// function takes the condition.text from API aas a paramter, and checks for the included to condition, 
// and provides a currated icon.

export const getIconForCondition = (conditionText: string) => {
   if (conditionText.includes('rain') || conditionText.includes('thunder')) {
     return require('./icons/day/266.png'); 
   } else if (conditionText.includes('lightning')) {
     return require('./icons/day/389.png');
   } else if (conditionText.includes('cloudy') || conditionText.includes('Overcast')) {
     return require('./icons/day/119.png');
   } else if (conditionText.includes('partly') || conditionText.includes('patchy')) {
     return require('./icons/day/116.png');
   } else if (conditionText.includes('sunny')) {
     return require('./icons/day/113.png'); 
   } else if (conditionText.includes('snow') || conditionText.includes('icy')) {
     return require('./icons/day/338.png');
   } else {
     return require('./icons/day/113.png'); 
   }
 };
